Clase base que hereda de HttpBaseActionResult y utilizada para estructurar una Solicitud de Consulta y que sea auto-consultable a través del protocolo ODATA

A continuación se presenta un ejemplo con la firma definida segun definición para un endpoint de la API.

Uso Basico

using System; using System.Net.Http; using Gale.REST.Http; namespace API.Areas.Account { /// <summary> /// Account API (Securified) /// </summary> [Gale.Security.Oauth.Jwt.Authorize] public class AccountController : Gale.REST.RestController { /// <summary> /// Return OData Queryable Data for Account List /// </summary> /// <returns></returns> public IHttpActionResult Get() { return new HttpQueryableActionResult<Models.Account>(this.Request); } } } [TableAttribute(Name="dbo.TB_MAE_Usuario")] public class Account { private System.Guid _token; private System.DateTime _createdAt; private string _identifier; private System.Nullable<System.Guid> _photo; private string _email; private string _fullname; private System.DateTime _lastConnection; private int _active; [ColumnAttribute(Name="ENTI_Token", Storage="_token", DbType="UniqueIdentifier NOT NULL", IsPrimaryKey=true)] public System.Guid token { get { return this._token; } set { this._token = value; } } [ColumnAttribute(Name="ENTI_FechaCreacion", Storage="_createdAt", DbType="DateTime NOT NULL")] public System.DateTime createdAt { get { return this._createdAt; } set { this._createdAt = value; } } [ColumnAttribute(Name="ENTI_Identificador", Storage="_identifier", DbType="VarChar(200) NOT NULL", CanBeNull=false)] public string identifier { get { return this._identifier; } set { this._identifier = value; } } [ColumnAttribute(Name="ARCH_Token", Storage="_photo", DbType="UniqueIdentifier")] public System.Nullable<System.Guid> photo { get { return this._photo; } set { this._photo = value; } } [ColumnAttribute(Name="USUA_Email", Storage="_email", DbType="VarChar(100) NOT NULL", CanBeNull=false)] public string email { get { return this._email; } set { this._email = value; } } [ColumnAttribute(Name="USUA_NombreCompleto", Storage="_fullname", DbType="VarChar(250) NOT NULL", CanBeNull=false)] public string fullname { get { return this._fullname; } set { this._fullname = value; } } [ColumnAttribute(Name="USUA_UltimaConexion", Storage="_lastConnection", DbType="DateTime NOT NULL")] public System.DateTime lastConnection { get { return this._lastConnection; } set { this._lastConnection = value; } } [ColumnAttribute(Name="USUA_Activo", Storage="_active")] public int active { get { return this._active; } set { this._active = value; } } }
